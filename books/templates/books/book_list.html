from django.views.generic import ListView from .models import Book class BookListView(ListView): model = Book template_name = 'books/book_list.html' # твой шаблон context_object_name = 'page_obj' # используем page_obj, как у тебя в шаблоне paginate_by = 5 def get_queryset(self): query = self.request.GET.get('q') if query: return Book.objects.filter(title__icontains=query) return Book.objects.all() def get_context_data(self, **kwargs): context = super().get_context_data(**kwargs) context['query'] = self.request.GET.get('q', '') return context
